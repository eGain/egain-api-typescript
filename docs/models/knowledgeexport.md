# KnowledgeExport

## Example Usage

```typescript
import { KnowledgeExport } from "@egain/egain-api-typescript/models";

let value: KnowledgeExport = {
  portalID: "PROD-1000",
  language: {
    code: "en-US",
  },
  resourceTypes: [
    "articles",
  ],
  dataDestination: {
    destinationType: "AWS S3 bucket",
    path: "s3://amzn-s3-demo-bucket/mydeptfolder",
    sftpDetails: {
      username: "rtka-user",
      password: "xyz1234",
      host: "sftp.yourcompany.com",
    },
    s3Details: {
      region: "us-west-2",
      accessKey: "s3-access-key",
      secretKey: "s3-access-secret",
    },
  },
};
```
