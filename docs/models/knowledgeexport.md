# KnowledgeExport

## Example Usage

```typescript
import { KnowledgeExport } from "@egain/egain-api-typescript/models";

let value: KnowledgeExport = {
  portalID: "PROD-1000",
  language: {
    code: "en-US",
  },
  resourceTypes: [
    "articles",
  ],
  dataDestination: {
    destinationType: "AWS S3 bucket",
    path: "https://s3.us-west-2.amazonaws.com/amzn-s3-demo-bucket1",
    region: "us-west-2",
    credentials: {
      accessKey: "s3-access-user",
      secretKey: "s3-access-secret",
    },
  },
};
```
